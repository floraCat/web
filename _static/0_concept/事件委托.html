事件委托 ------------------------------------ * 利用冒泡的原理，把事件加到父级上，触发执行效果 好处： ------------------------------------ * 提高性能 * 新添加的元素，还会有之前的事件 (监听也可以做到，但多个监听会占用内存，用事件委托性能会更好) 事件源： ------------------------------------ * 不管在哪个事件中，只要你操作的那个元素就是事件源 * ie下事件源 : window.event.srcElement * 标准下事件源: window.event.target * 事件源的元素名: nodeName 实例： ------------------------------------ <ol><li>111</li><li>222</li><li>333</li></ol> for(var i=0; i < aLi.length; i++){ oLi[i].onclick=function{ alert(123); } } //把事件委托给父级ul oUl.onclick=function(){ //点击li时会冒泡到ul，完成同样的操作 alert(123); } ———————————————————————————————————————————————————————————— for(var i=0; i < aLi.length; i++){ oLi[i].onclick=function{ this.style.background="red"; } } //把事件委托给父级ul oUl.onclick=function(){ var ev=ev || window.event; var target=ev.target || ev.srcElement; //事件源 if(target.nodeName.toLowerCase()=="li"){ //事件源元素名 target.style.background="red"; } } ———————————————————————————————————————————————————————————— //点击oBtn添加新的li var iNow=3; oBtn.onclick=function(){ iNow++; var oLi=document.createElement("li"); oLi.innerHTML=111 * iNow; oUl.append(oLi); } //新添加的li依然有之前点击加背景色的事件效果 ————————————————————————————————————————————————————————————
