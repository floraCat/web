对象引用： ------------------------------------ * 对象和函数都是引用的关系 实例： ------------------------------------ /*赋值*/ var a=5; var b=a; b+=3; alert(b); //8 alert(a); //5 //变量a赋值给b,不影响a本身 ———————————————————————————————————————————————————————————— var a=[1,2,3]; var b=a; b.push(4); alert(b); //1,2,3,4 alert(a); //1,2,3,4 //数组对象a赋给了b,a和b共用一个引用地址 ———————————————————————————————————————————————————————————— var a=[1,2,3]; var b=a; b=[1,2,3,4]; alert(b); //1,2,3,4 alert(a); //1,2,3 //数组对象a赋给了b,而b又重新定义后不再与b共用一个引用地址 ———————————————————————————————————————————————————————————— //克隆对象 var obj={a:10}; function copy(obj){ //浅拷贝 var newObj={}; for(var attr in obj){ newObj[attr]=obj[attr]; //赋值，非引用关系 } return newObj; } var obj2=copy(obj); //克隆一个新的对象，对原有对象本身不影响 obj2.a=20; alert(obj.a); //10 ———————————————————————————————————————————————————————————— //克隆嵌套对象 var obj={a:{b:10}}; function deepCopy(obj){ //深拷贝 if(typeof obj!="object"){return obj;} var newObj={}; for(var attr in obj){ newObj[attr]=deepCopy(obj[attr]); //递归 } return newObj; } obj2=deepCopy(obj); obj2.a.b=20; alert(obj.a.b); //10 ————————————————————————————————————————————————————————————
